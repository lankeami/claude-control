# Chat UI + Transcript Integration Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace the prompt-card dashboard with a messaging-style chat UI that shows the full Claude Code conversation by reading the session transcript JSONL file from disk.

**Architecture:** Hooks send `transcript_path` when registering sessions. The Go server stores it and exposes a new `GET /api/sessions/{id}/transcript` endpoint that reads the JSONL, extracts text messages, and returns them as JSON. The frontend replaces prompt cards with chat bubbles (Claude left/gray, user right/blue) and uses a large textarea for responses.

**Tech Stack:** Go (JSONL parsing, file I/O), Alpine.js, vanilla CSS

**Spec:** `docs/superpowers/specs/2026-03-18-chat-ui-design.md`

Create `server/api/transcript.go`:

```go
package api

import (
	"bufio"
	"encoding/json"
	"net/http"
	"os"
)

type transcriptMessage struct {
	Role      string `json:"role"`
	Content   string `json:"content"`
	Timestamp string `json:"timestamp"`
}

type transcriptEntry struct {
	Type      string          `json:"type"`
	Timestamp string          `json:"timestamp"`
	Message   json.RawMessage `json:"message"`
}

type messageContent struct {
	Role    string          `json:"role"`
	Content json.RawMessage `json:"content"`
}

type contentBlock struct {
	Type string `json:"type"`
	Text string `json:"text"`
}

func (s *Server) handleGetTranscript(w http.ResponseWriter, r *http.Request) {
	id := r.PathValue("id")

	path, err := s.store.GetTranscriptPath(id)
	if err != nil {
		http.Error(w, `{"error":"session not found"}`, http.StatusNotFound)
		return
	}

	messages := []transcriptMessage{}

	if path == "" {
		w.Header().Set("Content-Type", "application/json")
		json.NewEncoder(w).Encode(messages)
		return
	}

	f, err := os.Open(path)
	if err != nil {
		// File doesn't exist or not readable — return empty
		w.Header().Set("Content-Type", "application/json")
		json.NewEncoder(w).Encode(messages)
		return
	}
	defer f.Close()

	scanner := bufio.NewScanner(f)
	scanner.Buffer(make([]byte, 0, 1024*1024), 10*1024*1024) // 10MB max line

	for scanner.Scan() {
		var entry transcriptEntry
		if err := json.Unmarshal(scanner.Bytes(), &entry); err != nil {
			continue
		}

		// Only process user and assistant messages
		if entry.Type != "user" && entry.Type != "assistant" {
			continue
		}

		var msg messageContent
		if err := json.Unmarshal(entry.Message, &msg); err != nil {
			continue
		}

		// Extract text from content blocks
		text := extractText(msg.Content)
		if text == "" {
			continue // Skip entries with no text (tool_use only, tool_result only)
		}

		messages = append(messages, transcriptMessage{
			Role:      entry.Type,
			Content:   text,
			Timestamp: entry.Timestamp,
		})
	}

	// Safety cap: last 500 messages
	if len(messages) > 500 {
		messages = messages[len(messages)-500:]
	}

	w.Header().Set("Content-Type", "application/json")
	json.NewEncoder(w).Encode(messages)
}

func extractText(raw json.RawMessage) string {
	// Try as array of content blocks
	var blocks []contentBlock
	if err := json.Unmarshal(raw, &blocks); err == nil {
		var texts []string
		for _, b := range blocks {
			if b.Type == "text" && b.Text != "" {
				texts = append(texts, b.Text)
			}
		}
		if len(texts) > 0 {
			result := texts[0]
			for i := 1; i < len(texts); i++ {
				result += "\n" + texts[i]
			}
			return result
		}
		return ""
	}

	// Try as plain string
	var s string
	if err := json.Unmarshal(raw, &s); err == nil {
		return s
	}

	return ""
}
```

```js
    async fetchTranscript(sessionId) {
      if (!sessionId) {
        this.chatMessages = [];
        return;
      }
      this.chatLoading = true;
      try {
        const resp = await fetch(`/api/sessions/${sessionId}/transcript`, {
          headers: { 'Authorization': `Bearer ${this.apiKey}` }
        });
        if (resp.status === 401) { this.disconnect(); return; }
        if (resp.ok) {
          this.chatMessages = await resp.json();
          this.$nextTick(() => this.scrollToBottom());
        }
      } catch (e) {}
      this.chatLoading = false;
    },

    scrollToBottom() {
      const el = document.getElementById('chat-area');
      if (el) el.scrollTop = el.scrollHeight;
    },
```

```html
                <p style="font-size:1.125rem">No messages yet</p>
                <p style="font-size:0.875rem;margin-top:0.5rem">Conversation will appear here</p>
              </div>
            </div>
          </template>
          <template x-for="(msg, idx) in chatMessages" :key="idx">
            <div class="chat-bubble" :class="[msg.role, idx === chatMessages.length - 1 && currentPendingPrompt ? 'waiting' : '']">
              <span x-text="msg.content"></span>
              <span class="bubble-time" x-text="timeAgo(msg.timestamp)"></span>
            </div>
          </template>
        </div>

        <!-- Response textarea (only when pending prompt) -->
        <div class="response-area" x-show="currentPendingPrompt">
          <textarea x-model="responseText" placeholder="Type your response..."
                    @keydown.enter.prevent="if(!$event.shiftKey && responseText.trim()){
                      responseSending=true;
                      respondToPrompt(currentPendingPrompt.id, responseText).then(() => responseSending=false);
                    }"
                    :disabled="responseSending"></textarea>
          <button class="send-btn" :disabled="!responseText.trim() || responseSending"
                  @click="responseSending=true;respondToPrompt(currentPendingPrompt.id, responseText).then(() => responseSending=false)">
            &#9654;
          </button>
        </div>
```
